$ tput setaf <色番号> #文字色を色番号にする
$ tput setab <色番号> #文字色を色番号にする
$ tput cup <y 座標> <x 座標> # 上から x 行目左から y 文字目にカーソル移動
$ tput bold #太字
$ tput clear # clear コマンドと同じ効果
$ tput sgr0 # 装飾解除
$ tput cols # ターミナルの横幅を文字数で出力
$ tput lines # ターミナルの縦幅を文字数で出力
$ tput civis # カーソル非表示
$ tput cnorm # カーソル表示
